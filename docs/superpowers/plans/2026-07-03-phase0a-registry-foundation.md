# Phase 0A: Test Harness, Shared Types, and Model Registry — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Establish the Vitest TDD harness and a typed, normalized Model Registry, built from a pinned TokenCost snapshot, that every later Phase 0 subsystem (tokenizer, caching, cost core) consumes.

**Architecture:** A build-time script fetches a commit-pinned TokenCost JSON, a pure normalization layer converts each raw entry into a typed `ModelRecord` keyed on `(canonicalId, deployment)`, and a small query API loads the bundled artifact. All normalization is pure functions with unit tests written first. No React, no network at runtime.

**Tech Stack:** TypeScript 5.6 (strict), Vite 6, Vitest, tsx (for build scripts). Data source: `AgentOps-AI/tokencost` `tokencost/model_prices.json`.

## Global Constraints

Copied verbatim from the spec (v1.2). Every task inherits these.

- Fully client-side, no backend, no runtime data fetch by default.
- Dependencies pinned to exact versions (no `^` or `~`), committed lockfile, `npm ci` in CI.
- TypeScript strict mode; no implicit any; strict null checks.
- TDD mandatory: no engine function is written before its test. Red → green → refactor.
- Registry primary key is `(canonicalId, deployment)`. Divergent-price SKUs across deployments are kept, not collapsed.
- Billing units: `per_token | per_character | per_second | dbu`. Non-token units are never coerced to `$0`.
- Pricing snapshot pinned to a specific upstream commit SHA, not `main`.
- Coverage is a projection, never labeled "measured."
- Commits: the git author and committer stay the human; no AI attribution anywhere in commit messages.
- Commit after every green test (frequent commits).

Create `src/registry/resolveFamily.ts`. Order matters: `claude`/`gemini` run before `openai` so a family name inside a provider path cannot be hijacked, and open-family finetunes run before the generic OpenAI rule.

```ts
import type { TokenizerFamily, AccuracyTier } from '@/types/registry';

type Rule = { re: RegExp; family: TokenizerFamily; tier: AccuracyTier };

// First match wins. Applied to the lowercased id.
const RULES: Rule[] = [
  { re: /claude/, family: 'claude', tier: 'estimate' },
  { re: /gemini|gemma|(^|\/)palm|bison|unicorn|medlm|gecko/, family: 'gemini_gemma', tier: 'approx' },
  // open-family finetunes BEFORE the generic openai rule, so "...gpt4" in a Llama id does not hijack
  { re: /mistral|mixtral|ministral|codestral|pixtral|magistral|wizardlm|zephyr|openchat/, family: 'mistral', tier: 'exact_unverified' },
  { re: /llama|nemotron|hermes|mythomax|airoboros|remm-slerp|lzlv|firefunction|(^|\/)dolphin/, family: 'llama', tier: 'exact_unverified' },
  { re: /qwen|qwq|qwerky/, family: 'qwen', tier: 'exact_unverified' },
  { re: /deepseek/, family: 'deepseek', tier: 'exact_unverified' },
  { re: /kimi|moonshot/, family: 'kimi', tier: 'exact_unverified' },
  { re: /command|(^|\/)cohere|(^|\/)c4ai|(^|\/)embed-(english|multilingual)|embed-v[- ]?4/, family: 'cohere', tier: 'exact_unverified' },
  { re: /jamba|jurassic|(^|\/)j2-|ai21|jais/, family: 'ai21', tier: 'exact_unverified' },
  { re: /(^|\/)phi-|phi3|phi4|phi-4/, family: 'phi', tier: 'exact_unverified' },
  { re: /(^|\/)yi-|(^|\/)01-ai/, family: 'yi', tier: 'exact_unverified' },
  { re: /granite/, family: 'granite', tier: 'exact_unverified' },
  { re: /dbrx|databricks/, family: 'dbrx', tier: 'exact_unverified' },
  { re: /sonar|perplexity|(^|\/)pplx/, family: 'perplexity', tier: 'exact_unverified' },
  { re: /grok/, family: 'grok', tier: 'approx' },
  { re: /titan|amazon\.nova|amazon\.titan|(^|\/)nova($|[-/])/, family: 'amazon', tier: 'estimate' },
  { re: /voyage/, family: 'voyage', tier: 'estimate' },
  { re: /aleph|luminous/, family: 'aleph_alpha', tier: 'estimate' },
  // generic OpenAI LAST among named families so finetunes above win
  { re: /gpt|chatgpt|(^|\/)o1|(^|\/)o3|(^|\/)o4|davinci|curie|babbage|(^|\/)ada($|[-/])|text-embedding-(ada|3)|codex|computer-use/, family: 'openai', tier: 'exact_unverified' },
];

export function resolveFamily(id: string): { family: TokenizerFamily; tier: AccuracyTier } {
  const k = id.toLowerCase();
  for (const rule of RULES) {
    if (rule.re.test(k)) return { family: rule.family, tier: rule.tier };
  }
  return { family: 'unknown', tier: 'estimate' };
}
```

import type { BillingUnit } from '@/types/registry';

export type RawEntry = Record<string, unknown>;

const num = (v: unknown): number | null =>
  typeof v === 'number' && Number.isFinite(v) ? v : null;

const PER_MILLION = 1_000_000;

export function detectBillingUnit(e: RawEntry): BillingUnit {
  if (num(e.input_cost_per_character) !== null) return 'per_character';
  if (num(e.input_cost_per_second) !== null && num(e.input_cost_per_token) === null) return 'per_second';
  if (num(e.input_dbu_cost_per_token) !== null && num(e.input_cost_per_token) === null) return 'dbu';
  return 'per_token';
}

// Token and character prices scale to per-million; per_second and dbu are stored raw.
export function normalizeInputPrice(e: RawEntry, unit: BillingUnit): number | null {
  switch (unit) {
    case 'per_token': { const v = num(e.input_cost_per_token); return v === null ? null : v * PER_MILLION; }
    case 'per_character': { const v = num(e.input_cost_per_character); return v === null ? null : v * PER_MILLION; }
    case 'per_second': return num(e.input_cost_per_second);
    case 'dbu': return num(e.input_dbu_cost_per_token);
  }
}

export function normalizeOutputPrice(e: RawEntry): number | null {
  const v = num(e.output_cost_per_token);
  return v === null ? null : v * PER_MILLION;
}

// The deployment is the routing prefix (everything up to the model segment) or the
// litellm_provider for a bare key. canonicalId is the model segment (deployment stripped).
export function parseKey(
  rawKey: string,
  e: RawEntry,
): { canonicalId: string; deployment: string; provider: string } {
  const provider = typeof e.litellm_provider === 'string' ? e.litellm_provider : 'unknown';
  const slash = rawKey.indexOf('/');
  if (slash === -1) {
    return { canonicalId: rawKey, deployment: provider, provider };
  }
  const lastSlash = rawKey.lastIndexOf('/');
  const deployment = rawKey.slice(0, lastSlash);
  const canonicalId = rawKey.slice(lastSlash + 1);
  // Two-segment ids like "deepseek/deepseek-r1" under openrouter keep provider/model as canonical.
  const parts = rawKey.split('/');
  if (parts.length >= 3) {
    return { canonicalId: parts.slice(-2).join('/'), deployment: parts[0], provider };
  }
  return { canonicalId, deployment, provider };
}

import type { PriceTier } from '@/types/registry';

const PER_M = 1_000_000;
const numOrU = (v: unknown): number | undefined => {
  const n = typeof v === 'number' && Number.isFinite(v) ? v : undefined;
  return n === undefined ? undefined : n * PER_M;
};

export function parseTiers(e: RawEntry): PriceTier[] {
  const tiers: PriceTier[] = [];
  for (const t of [128000, 200000] as const) {
    const inp = numOrU(e[`input_cost_per_token_above_${t}_tokens`]);
    const out = numOrU(e[`output_cost_per_token_above_${t}_tokens`]);
    const cr = numOrU(e[`cache_read_input_token_cost_above_${t}_tokens`]);
    const cw = numOrU(e[`cache_creation_input_token_cost_above_${t}_tokens`]);
    if (inp === undefined && out === undefined && cr === undefined && cw === undefined) continue;
    tiers.push({
      thresholdTokens: t,
      inputPrice: inp ?? 0,
      outputPrice: out ?? null,
      ...(cr !== undefined ? { cacheReadPerMToken: cr } : {}),
      ...(cw !== undefined ? { cacheWritePerMToken: cw } : {}),
    });
  }
  return tiers;
}

import type { CacheSpec, CacheArchetype } from '@/types/registry';

const BREAKPOINT_PROVIDERS = /anthropic|bedrock/;
const STORAGE_PROVIDERS = /gemini|vertex/;

function archetypeFor(provider: string): CacheArchetype {
  if (BREAKPOINT_PROVIDERS.test(provider)) return 'breakpoint_ttl';
  if (STORAGE_PROVIDERS.test(provider)) return 'storage';
  return 'automatic';
}

export function resolveCacheSpec(e: RawEntry, provider: string): CacheSpec | null {
  const read = num(e.cache_read_input_token_cost) ?? num(e.input_cost_per_token_cache_hit);
  const write = num(e.cache_creation_input_token_cost);
  const supported = e.supports_prompt_caching === true || read !== null || write !== null;
  if (!supported) return null;
  const archetype = archetypeFor(provider);
  if (read === null && write === null) return { archetype, rateUnavailable: true };
  return {
    archetype,
    ...(read !== null ? { cacheReadPerMToken: read * PER_MILLION } : {}),
    ...(write !== null ? { cacheWritePerMToken: write * PER_MILLION } : {}),
    rateUnavailable: false,
  };
}

// Drop only junk rows (no mode = a pricing bucket, not a model). A priced-at-zero
// model with a valid mode is a free model, kept and flagged.
export function classifyRow(canonicalId: string, e: RawEntry): { drop: boolean; freeTier: boolean } {
  const hasMode = typeof e.mode === 'string';
  if (!hasMode) return { drop: true, freeTier: false };
  const inTok = num(e.input_cost_per_token);
  const outTok = num(e.output_cost_per_token);
  const hasAnyUnit =
    inTok !== null || num(e.input_cost_per_character) !== null ||
    num(e.input_cost_per_second) !== null || num(e.input_dbu_cost_per_token) !== null;
  const freeTier = inTok === 0 && (outTok === 0 || outTok === null);
  if (!hasAnyUnit) return { drop: true, freeTier: false };  // price-null junk
  return { drop: false, freeTier };
}

import type { ModelRecord } from '@/types/registry';
import { resolveFamily } from '@/registry/resolveFamily';

export function normalizeEntry(rawKey: string, e: RawEntry): ModelRecord | null {
  const { canonicalId, deployment, provider } = parseKey(rawKey, e);
  const { drop, freeTier } = classifyRow(canonicalId, e);
  if (drop) return null;
  const billingUnit = detectBillingUnit(e);
  const inputPrice = normalizeInputPrice(e, billingUnit);
  if (inputPrice === null) return null;
  const { family, tier } = resolveFamily(canonicalId);
  const mode = e.mode as ModelRecord['mode'];
  const reasoning = num(e.output_cost_per_reasoning_token);
  return {
    canonicalId,
    deployment,
    displayName: canonicalId,
    provider,
    underlyingFamily: family,
    mode,
    billingUnit,
    inputPrice,
    outputPrice: mode === 'embedding' ? null : normalizeOutputPrice(e),
    reasoningPerMToken: reasoning === null ? null : reasoning * PER_MILLION,
    cache: resolveCacheSpec(e, provider),
    contextWindow: num(e.max_input_tokens),
    maxOutput: num(e.max_output_tokens),
    tiers: parseTiers(e),
    accuracyTier: tier,
    freeTier,
    deprecated: e.deprecated === true,
  };
}

export function normalizeCatalog(
  raw: Record<string, RawEntry>,
): { models: ModelRecord[]; droppedCount: number } {
  const models: ModelRecord[] = [];
  let droppedCount = 0;
  for (const [key, entry] of Object.entries(raw)) {
    const rec = normalizeEntry(key, entry);
    if (rec === null) droppedCount++;
    else models.push(rec);
  }
  return { models, droppedCount };
}

// Key includes deployment AND price, so different-price SKUs are never collapsed.
export function dedupeRecords(models: ModelRecord[]): ModelRecord[] {
  const seen = new Map<string, ModelRecord>();
  for (const m of models) {
    const key = `${m.canonicalId}|${m.deployment}|${m.inputPrice}|${m.outputPrice}`;
    if (!seen.has(key)) seen.set(key, m);
  }
  return [...seen.values()];
}

  // Date is passed by the caller/CI env to keep the build reproducible; no Date.now() in engine code.
  const date = process.env.SNAPSHOT_DATE ?? '1970-01-01';
  const snap = buildSnapshot(raw, PINNED_COMMIT, date);
  writeFileSync(
    new URL('../../src/config/registry.generated.json', import.meta.url),
    JSON.stringify(snap),
  );
  console.log(`registry: ${snap.models.length} models, ${snap.droppedCount} dropped`);
}

// Only run when invoked directly, not when imported by a test.
if (import.meta.url === `file://${process.argv[1]}`) {
  main().catch((err) => {
    console.error(err);
    process.exit(1);
  });
}

import type { RegistrySnapshot, ModelRecord } from '@/types/registry';

let current: RegistrySnapshot | null = null;
const byKey = new Map<string, ModelRecord>();
const byCanonical = new Map<string, ModelRecord[]>();

const key = (id: string, deployment: string): string => `${id}|${deployment}`;

export function loadRegistry(snap: RegistrySnapshot): void {
  current = snap;
  byKey.clear();
  byCanonical.clear();
  for (const m of snap.models) {
    byKey.set(key(m.canonicalId, m.deployment), m);
    const list = byCanonical.get(m.canonicalId) ?? [];
    list.push(m);
    byCanonical.set(m.canonicalId, list);
  }
}

export function getModel(id: string, deployment: string): ModelRecord | null {
  return byKey.get(key(id, deployment)) ?? null;
}

export function getDeployments(canonicalId: string): ModelRecord[] {
  return byCanonical.get(canonicalId) ?? [];
}

export function listByMode(mode: ModelRecord['mode']): ModelRecord[] {
  return (current?.models ?? []).filter((m) => m.mode === mode);
}

export function getSnapshotMeta(): { snapshotVersion: string; snapshotDate: string; droppedCount: number } {
  if (!current) throw new Error('registry not loaded');
  const { snapshotVersion, snapshotDate, droppedCount } = current;
  return { snapshotVersion, snapshotDate, droppedCount };
}
